### Jio ###

#1
Jio Churned Customer = COUNTROWS(FILTER(telecom_churn,telecom_churn[Reliance_Jio_Churned]=1))

#2
Jio Total Customers = COUNTROWS(FILTER(telecom_churn,telecom_churn[telecom_partner] = "Reliance Jio"))

#3
Jio Stayed Customers = 'Jio Measures Table'[Jio Total Customers] - 'Jio Measures Table'[Jio Churned Customer]

#4
Total Calls Made in Jio = CALCULATE(SUM(telecom_churn[Calls_Made_p] ),telecom_churn[telecom_partner] = "Reliance Jio")

#5 
Total SMS Sent in Jio = CALCULATE(SUM(telecom_churn[SMS_Sent_p]),telecom_churn[telecom_partner] = "Reliance Jio")

#6
Total Data Used By Customer in Jio = CALCULATE(SUM(telecom_churn[Data_used_p] ),telecom_churn[telecom_partner] = "Reliance Jio")

#7
AVG Calls Made By pc in Jio = CALCULATE(AVERAGE(telecom_churn[Calls_Made_p] ),telecom_churn[telecom_partner] = "Reliance Jio")

#8
AVG SMS Sent  by pc in Jio = CALCULATE(AVERAGE(telecom_churn[SMS_Sent_p]),telecom_churn[telecom_partner] = "Reliance Jio")

#9
AVG Data Used By Customer in Jio = CALCULATE(AVERAGE(telecom_churn[Data_used_p] ),telecom_churn[telecom_partner] = "Reliance Jio")

#10
MostFrequentAgeJio = 
VAR AgeWithCount =
    SUMMARIZE(
        FILTER(telecom_churn, telecom_churn[telecom_partner] = "Reliance Jio"),
        telecom_churn[Age],
        "AgeCount", COUNT(telecom_churn[Age])
    )
VAR MaxAgeCount =
    MAXX(AgeWithCount, [AgeCount])
VAR Result =
    CALCULATE(
        MAX(telecom_churn[Age]),
        FILTER(AgeWithCount, [AgeCount] = MaxAgeCount)
    )
RETURN Result
